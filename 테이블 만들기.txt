쿼리박스 ->books 계정 연결해서 밑에 적기

1.
create table login 
(userId varchar(30), 
userName varchar(20) not null,
userPwd varchar(20) not null,
userEmail varchar(50),
userBirth date,
userTel varchar(20) not null,
userNick varchar(20) ,
gender varchar(10),
constraint login_userId_pk primary key(userId),
constraint login_userNick_uk UNIQUE (userNick));

2.
create table review
(userId varchar(30) ,
userNick varchar(20),
content varchar(100),
reviewDate DATE DEFAULT sysdate,
grade varchar(20),
constraint review_userId_fk foreign key(userId) REFERENCES login(userId),
constraint review_userNick foreign key(userNick) REFERENCES login(userNick));

3.
create table krbook
(ISBN varchar(30),
subject varchar(30),
category1 varchar(30),
category2 varchar(30),
author varchar(20),
bookdate varchar(20),
page NUMBER,
booksize varchar(30),
price NUMBER,
constraint krbook_ISBN_pk primary key (ISBN));

ALTER TABLE krbook
ADD publisher varchar(20);

ALTER TABLE krbook
ADD translation varchar(20);

4.
create table author
(authorId VARCHAR(30),
authorName varchar(20),
authorBirth varchar(20),
authorPlace varchar(20),
authorDebut varchar(30),
debutBook varchar(30),
authorCategory varchar(20),
CONSTRAINT author_authorId_pk PRIMARY KEY(authorId));

5. 
CREATE TABLE authoretc
(authorId VARCHAR(30),
authorcontent varchar(100),
academic VARCHAR(30),
CONSTRAINT authoretc_authorId_fk FOREIGN KEY(authorId) REFERENCES author(authorId));
